[CERC1996] 机器人搬重物

题目描述

机器人移动学会（RMI）现在正尝试用机器人搬运物品。机器人的形状是一个直径 1.6 米的球。在试验阶段，机器人被用于在一个储藏室中搬运货物。储藏室是一个 N×M 的网格，有些格子为不可移动的障碍。机器人的中心总是在格点上，当然，机器人必须在最短的时间内把物品搬运到指定的地方。机器人接受的指令有：

  • 向前移动 1 步（Creep）；

  • 向前移动 2 步（Walk）；

  • 向前移动 3 步（Run）；

  • 向左转（Left）；

  • 向右转（Right）。

每个指令所需要的时间为 1 秒。请你计算一下机器人完成任务所需的最少时间。

思路分析

复杂网格图上问题首选 bfs 。给出的网格图下标从 (0,0)→(n−1,m−1) ，而机器人是运动在格点上且不能碰到地图边界，说明机器人运动过程中其中心坐标 (x,y) 必须满足相邻上下左右四个格子都无障碍；我们将每个指令都对应为广搜的一个 step ，引入三维张量 (x,y,direc) 描述处于位置 (x,y) 、朝向为 direc 的最少步数。剩下就是广搜模板，实现中一些细节还需注意。

#include<bits/stdc++.h>
using namespace std;
const int N=55;           
int n,m;
bool block[N][N],vis[N][N][4];       
//0-东(E)，1-南(S)，2-西(W)，3-北(N)
int dx[]={0,1,0,-1};
int dy[]={1,0,-1,0};
int getdirec(char c){
    if(c=='E') return 0;
    if(c=='S') return 1;
    if(c=='W') return 2;
    if(c=='N') return 3;
    return -1;
}
struct Node{
    int x,y,dir,step;
    Node(int _x,int _y,int _dir,int _step):x(_x),y(_y),dir(_dir),step(_step){}
};
int bfs(int sx,int sy,int ex,int ey,int direc){
    queue<Node> q;
    q.push(Node(sx,sy,direc,0));
    vis[sx][sy][direc]=true;
    while(!q.empty()){
        Node cur=q.front();
        q.pop();
        if(cur.x==ex && cur.y==ey)
            return cur.step;
        // 左转
        int new_dir=(cur.dir+3)%4;   // 左转 = 方向索引 -1，模4处理
        if(!vis[cur.x][cur.y][new_dir]){
            vis[cur.x][cur.y][new_dir]=true;
            q.push(Node(cur.x,cur.y,new_dir,cur.step+1));
        }
        // 右转
        new_dir=(cur.dir+1)%4;       // 右转 = 方向索引 +1
        if(!vis[cur.x][cur.y][new_dir]){
            vis[cur.x][cur.y][new_dir]=true;
            q.push(Node(cur.x,cur.y,new_dir,cur.step+1));
        }
        // 前进
        for(int st=1;st<=3;st++){
            int nx=cur.x+dx[cur.dir]*st;
            int ny=cur.y+dy[cur.dir]*st;
            if(nx<=0 || nx>=n || ny<=0 || ny>=m) break; 
            if(block[nx-1][ny-1] || block[nx-1][ny] || block[nx][ny-1] || block[nx][ny])
                break; 
            bool ok=1;
            for(int s=1;s<=st;++s){
                int tx=cur.x+dx[cur.dir]*s;
                int ty=cur.y+dy[cur.dir]*s;
                if(block[tx-1][ty-1] || block[tx-1][ty] || block[tx][ty-1] || block[tx][ty]){
                    ok=false;
                    break;
                }
            }
            if(!ok)break;
            if(!vis[nx][ny][cur.dir]){
                vis[nx][ny][cur.dir]=true;
                q.push(Node(nx,ny,cur.dir,cur.step+1));
            }
        }
    }
    return -1; 
}
int main(){
    cin>>n>>m;
    for(int i=0;i<n;i++){
        for(int x,j=0;j<m;j++){
            cin>>x;
            block[i][j]=(x==1); 
        }
    }
    int stx,sty,edx,edy;char c;
    cin>>stx>>sty>>edx>>edy>>c;
    int direc=getdirec(c);
    int ans=bfs(stx,sty,edx,edy,direc);
    cout<<ans;
    return 0;
}

 

[ICPC 2010 Chengdu R] Error Curves

题目描述

给定 n 个二次函数 f1(x),f2(x),…,fn(x)（均形如 ax2+bx+c,a≥0），设 F(x)=max{f1(x),f2(x),...,fn(x)}，求 F(x) 在区间 [0,1000] 上的最小值。

思路分析

对于定义在 D 上的连续单谷函数 f(x),g(x) ，不难证明 h(x)=max{f(x),g(x)} 为单谷函数。进而由归纳法知道：

F(x)=max{f1(x),f2(x),...,fn(x)}

为单谷函数。因此可以考虑三分法：即对于区间 [l,r] ，考虑其左侧 1/3 处的 F 值与右侧 1/3 处的 F 值，更小的值必然更接近谷底，依此缩小区间。

Code

xxxxxxxxxx
#include<bits/stdc++.h>
using namespace std;
//二分对单调函数求极值近似解，三分对单峰函数也有同样效果
const int N=1e5+10,inf=0x3f3f3f3f;
const double eps=1e-9;
int n,a[N],b[N],c[N];
double f(double x){
    double maxn=-inf;
    for(int i=1;i<=n;i++)
        maxn=max(maxn,a[i]*x*x+b[i]*x+c[i]);
    return maxn;
}
int main(){
    int t;cin>>t;
    while(t--){
        cin>>n;
        for(int i=1;i<=n;i++)
            cin>>a[i]>>b[i]>>c[i];
        double l=0,r=1000;
        while(r-l>eps){
            double mid1=(2*l+r)/3,mid2=(2*r+l)/3;
            if(f(mid1)<f(mid2))
                r=mid2;
            else
                l=mid1;
        }
        printf("%.4lf\n",f(l));
    }
    return 0;
}

[XJTUPC 2024] 雪中楼

题目描述

西安交通大学下雪了，所有楼顶都被积雪覆盖。

在梧桐道上有 n 座房子，自南向北第 i 座房子的标号是 i，所有房子的高度互不相同。LNC 从南向北，依次登上每栋楼楼顶，然后他会回头观察他已经到达过的房子，并记录下这些房子中，比当前他所在房子矮的所有房子中，最高的房子的编号，记为 ai。特殊的，当没有房子比当前房子矮的时候， ai=0。

现在给你所有的 ai，要求你还原出所有楼的大小关系。由于高度的具体值未知，你只需要按照高度从低到高，输出房子编号。

可以证明对于任何合法的输入，均存在唯一解。

思路分析

核心洞察：从矮到高排序的序列中，ai 必须在 i 的前面。将 ai=0 视为空地，则我们得到了若干偏序关系，现在期望得到全序。自然地想到拓扑排序。

我们对每个 (ai,i) 建立有向边，在形成的 DAG 上跑拓扑排序即可。然而可能会有疑惑：既然拓扑排序是求解全序的通用算法，题目中为何要保证 ai 的最大性？

[ICPC 2022 Tehran R] Parking Party

题目描述

Peyman 决定在他位于扎博尔（Zabol）的家中举办一场晚宴派对。他的房子有一个矩形的停车场，可以用一个 n 行 m 列的网格来表示。一辆汽车可以停在这个网格的 n×m 个单元格之一。然而，一些单元格被不可移动的柱子占据，因此没有汽车可以停在其中或穿过它们。停车场的每一行或每一列在其两端都有一个入口。当一辆汽车通过一个入口进入停车场时，它只能沿着对应的行或列直线前进；当它到达对面的入口，或者下一个单元格被柱子或另一辆已停放的汽车占据时，它会停下来并停在一个网格单元格中。此外，如果某行或某列的第一个单元格已经被占据，汽车就不能从该行或该列的入口进入。

Peyman 希望最大化可以停在他停车场中的汽车数量。为了实现这一目标，他可以指示客人在到达时从哪个入口进入。你的任务是帮助 Peyman 完成这个任务。

思路分析

定义好方向数组后分别从 (i,1),(i,m),(1,i),(n,i) 开始沿着对应方向搜就行了。

Code

xxxxxxxxxx
#include<bits/stdc++.h>
using namespace std;
const int N=1e3+10;
char a[N][N];
int n,m,dy[]={1,-1,0,0},dx[]={0,0,1,-1};
bool vis[N][N];
void dfs(int x,int y,int dir){
    if(x<1 || y<1 || x>n || y>m || a[x][y]=='o')
        return ;
    if(!vis[x][y])vis[x][y]=1;
    dfs(x+dx[dir],y+dy[dir],dir);
}
int main(){
    cin>>n>>m;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            cin>>a[i][j];
    for(int i=1;i<=n;i++)
        dfs(i,1,0),dfs(i,m,1);
    for(int i=1;i<=m;i++)
        dfs(1,i,2),dfs(n,i,3);
    int ans=0;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            ans+=vis[i][j];
    cout<<ans;
    return 0;
}

[JSOI2008] 球形空间产生器

题目描述

有一个球形空间产生器能够在 n 维空间中产生一个坚硬的球体。现在，你被困在了这个 n 维球体中，你只知道球面上 n+1 个点的坐标，你需要以最快的速度确定这个 n 维球体的球心坐标，以便于摧毁这个球形空间产生器。

思路分析

给出 n 维空间上的 n+1 个点的坐标，求到它们欧氏距离相等的点坐标。

这是一个 n 元方程组问题，即求解向量 x 使：

∑j=1n(ai,j−xj)2=r2

∑j=1n[(ai,j−xj)2−(ai+1,j−xj)2]=0

也即：

∑j=1n2(ai,j−ai+1,j)xj=∑j=1n(ai+1,j−ai,j)2

其中 i∈{1,2,…,n}。命系数矩阵 A 为：

A=(2(a1,1−a2,1)2(a1,2−a2,2)⋯2(a1,n−a2,n)2(a2,1−a3,1)2(a2,2−a3,2)⋯2(a2,n−a3,n)⋮⋮⋱⋮2(an,1−an+1,1)2(an,2−an+1,2)⋯2(an,n−an+1,n))

对应解向量为：

b=(∑j=1n(a2,j−a1,j)2∑j=1n(a3,j−a2,j)2⋮∑j=1n(an+1,j−an,j)2)

于是有线性方程组 Ax=b​ 。这就是一个 n 元齐次线性方程组了，可以用高斯消元法求解。

Code

xxxxxxxxxx
#include<bits/stdc++.h>
using namespace std;
const double eps=1e-8;
const int N=12;
int n;double a[N][N],c[N][N],b[N];
//c为系数阵，b为常数向量。
int main(){
    cin>>n;
    for(int i=1;i<=n+1;i++)
        for(int j=1;j<=n;j++)
            cin>>a[i][j];
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++){
            c[i][j]=2*(a[i][j]-a[i+1][j]);
            b[i]+=a[i][j]*a[i][j]-a[i+1][j]*a[i+1][j];
        }
    }
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++){
            if(fabs(c[i][j])>eps){
                swap(c[i],c[j]);
                swap(b[i],b[j]);
            }
        }
        for(int j=1;j<=n;j++){
            if(i==j)continue;
            double r=c[j][i]/c[i][i];
            for(int k=i;k<=n;++k)
                c[j][k]-=c[i][k]*r;
            b[j]-=b[i]*r;
        }
    }
    for(int i=1;i<=n;++i)
        printf("%.3lf ",b[i]/c[i][i]);
    return 0;
}

可达性统计

题目描述

给定一张 N 个点 M 条边的有向无环图，统计从每个点出发能到达的点的数量。

思路分析

自底向上的顺序结构：在树上表现为从叶子节点->每个子树->根

在 DAG 上表现为从拓扑序上最后的节点->拓扑序上最前的节点

对每个节点 i ，不妨用一个 n 位二进制数 fi 表达其对于其它节点的可达性。这个更新显然是自底向上的。所以建图后跑一遍拓扑排序，倒序遍历节点 x。对每个 x→y ，y 可达的 x 必然可达，所以需要通过 f[x] = f[x] | f[y]来更新 fx 的值。时间复杂度 O(N(N+M)/32)，空间复杂度 O(n2) 。当然，对于 N 较小的稠密图，可以直接使用 Floyd 算法求传递闭包，用 bitset 优化后时间复杂度 O(n3/32)。

Code

xxxxxxxxxx
#include<bits/stdc++.h>
using namespace std;
//细节 DAG
const int N=3e4+10;
int n,m,head[N],ver[N],nxt[N],tot;
//head[u]:以u起点的最后一条加入边的编号
//ver[i]:第 i 条边的终点
//nxt[i]:与第 i 条边同一起点的下一条边的编号
void add(int x,int y){
    ver[++tot]=y;
    nxt[tot]=head[x];
    head[x]=tot;
}
int deg[N],tp[N],cnt;
void topusort(){
    queue<int> q;
    for(int i=1;i<=n;i++)
        if(!deg[i])q.push(i);
    while(!q.empty()){
        int x=q.front();q.pop();
        tp[++cnt]=x;
        for(int i=head[x];i;i=nxt[i]){
            int y=ver[i];
            if(!(--deg[y]))
                q.push(y);
        }
    }
}
bitset<N> f[N];
int main(){
    cin>>n>>m;
    for(int i=1,x,y;i<=m;i++){
        cin>>x>>y;
        add(x,y);
        ++deg[y];
    }
    topusort();
    for(int i=cnt;i>=1;i--){
        int x=tp[i];
        f[x][x]=1;
        for(int i=head[x];i;i=nxt[i]){
            int y=ver[i];
            f[x]|=f[y];
        }
    }
    for(int i=1;i<=n;i++)
        cout<<f[i].count()<<endl;
    return 0;
}
